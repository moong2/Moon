### C++
{% highlight css %}
#include <iostream>
using namespace std;

int main()
{
	int T, a, b, data;
	cin >> T;
	for (int i = 0; i < T; i++)
	{
		cin >> a >> b;
		data = 1;
		for (int j = 0; j < b; j++) data = data * a % 10;
		if (data == 0) data = 10;
		cout << data << endl;
	}
}
{% endhighlight %}
