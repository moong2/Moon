### C++
{% highlight css %} 
#include <iostream>
using namespace std;

int main()
{
	int number[5];
	int UniqueNum = 0;
	for (int i = 0; i < 5; i++)
	{
		cin >> number[i];
		UniqueNum += (number[i] * number[i]);
	}
	UniqueNum %= 10;
	cout << UniqueNum;
}
