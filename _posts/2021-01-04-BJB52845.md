### C++
{% highlight css %} 
#include <iostream>
using namespace std;

int main()
{
	int L, P, article[5];
	int people;
	cin >> L >> P;
	people = L * P;
	for (int i = 0; i < 5; i++) 
	{
		cin >> article[i];
	}
	for (int i = 0; i < 5; i++)
	{
		cout << -(people - article[i]) << " ";
	}
}

L, M = map(int, input().split())
