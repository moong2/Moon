### C++
{% highlight css %}
#include <iostream>
using namespace std;

int main()
{
	int x, y, w, h, minx, miny;
	cin >> x >> y >> w >> h;
	minx = (x > w - x ? w - x : x);
	miny = (y > h - y ? h - y : y);
	cout << (minx < miny ? minx : miny);
}
