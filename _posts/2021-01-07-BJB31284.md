int blankSpace(int n)
{
	int a = n, b, sum = 2;
	while (true)
	{
		b = a % 10;
		a /= 10;
		if (b == 0) sum += 4;
		else if (b == 1) sum += 2;
		else sum += 3;
		if (a == 0) break;
		sum++;
	}
	return sum;
}

def calculateBlankSpace(n) :
    a = n
    sum = 2
    while(1) :
        b = a % 10
        a //= 10
        if(b == 0) : sum += 4
        elif (b == 1) : sum += 2
        else : sum += 3
        if(a == 0) : break
        sum += 1
    return sum
